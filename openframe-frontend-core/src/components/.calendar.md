A pre-styled calendar component built on top of react-day-picker with custom Tailwind CSS styling for date selection interfaces.

## Key Components

- **Calendar**: Main component that wraps DayPicker with predefined styles
- **CalendarProps**: TypeScript type extending DayPicker's component props
- Custom styling for months, navigation, cells, and day states (selected, today, disabled, etc.)
- Navigation arrows using Lucide React icons (ChevronLeft, ChevronRight)

## Usage Example

```typescript
import { Calendar } from "./components/calendar"
import { useState } from "react"

function DatePicker() {
  const [date, setDate] = useState<Date | undefined>(new Date())

  return (
    <Calendar
      mode="single"
      selected={date}
      onSelect={setDate}
      className="rounded-md border"
      disabled={(date) =>
        date > new Date() || date < new Date("1900-01-01")
      }
    />
  )
}

// Range selection
function DateRangePicker() {
  const [dateRange, setDateRange] = useState<DateRange | undefined>()

  return (
    <Calendar
      mode="range"
      selected={dateRange}
      onSelect={setDateRange}
      numberOfMonths={2}
    />
  )
}
```
