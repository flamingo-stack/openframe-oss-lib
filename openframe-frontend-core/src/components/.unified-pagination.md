A Next.js client component that provides URL-aware pagination functionality while maintaining scroll position and supporting local state updates without full page reloads.

## Key Components

- **UnifiedPagination**: Main component that wraps the base Pagination component with URL management
- **handlePageChange**: Core function that updates local state, modifies URL parameters, and preserves scroll position
- **UnifiedPaginationProps**: Interface defining component props including page state and optional callback

## Usage Example

```typescript
import { UnifiedPagination } from './unified-pagination'

function ProductList() {
  const [currentPage, setCurrentPage] = useState(1)
  const [products, setProducts] = useState([])
  const totalPages = Math.ceil(products.length / itemsPerPage)

  const handlePageChange = (page: number) => {
    setCurrentPage(page)
    // Fetch new data or filter existing data
    fetchProducts(page)
  }

  return (
    <div>
      <ProductGrid products={currentProducts} />
      <UnifiedPagination
        currentPage={currentPage}
        totalPages={totalPages}
        onPageChange={handlePageChange}
        className="mt-6 flex justify-center"
      />
    </div>
  )
}
```

The component automatically updates the URL with `?page=X` parameters for bookmarking support while preventing page reloads and maintaining user scroll position.
