A React component for selecting organizations from a dropdown menu with pre-fetched images, loading states, and configurable styling using the ODS design system.

## Key Components

- **OrganizationSelector** - Main dropdown component with organization selection functionality
- **OrganizationOption** - Type interface defining organization data structure  
- **OrganizationSelectorProps** - Props interface with extensive configuration options
- **useBatchImages hook** - Pre-fetches organization images for better performance
- **OrganizationIcon component** - Displays organization logos with fallback to initials

## Usage Example

```typescript
import { OrganizationSelector, OrganizationOption } from './organization-selector'
import { useState } from 'react'

const organizations: OrganizationOption[] = [
  {
    id: '1',
    organizationId: 'org-123',
    name: 'Acme Corp',
    imageUrl: 'https://example.com/logo.png',
    isDefault: true
  },
  {
    id: '2', 
    organizationId: 'org-456',
    name: 'Tech Solutions Inc',
    imageUrl: 'https://example.com/logo2.png'
  }
]

function MyComponent() {
  const [selectedOrgId, setSelectedOrgId] = useState('')
  const [isLoading, setIsLoading] = useState(false)

  return (
    <OrganizationSelector
      organizations={organizations}
      value={selectedOrgId}
      onValueChange={setSelectedOrgId}
      label="Select Organization"
      placeholder="Choose your organization"
      iconSize="md"
      isLoading={isLoading}
      maxItems={50}
    />
  )
}
```
