A customizable button-based selector component for multi-select and single-select interfaces with visual feedback and validation states.

## Key Components

- **`SelectableOption`** - Interface defining selectable items with id, name, description, icon, and color properties
- **`PushButtonSelector`** - Main component that renders clickable option cards with selection indicators
- **`toggleSelection`** - Handles selection logic for both multi-select and single-select modes
- **`getSelectedOptions`** - Utility function to retrieve currently selected option objects

## Usage Example

```typescript
import { PushButtonSelector, SelectableOption } from './push-button-selector';
import { Database, Cloud, Shield } from 'lucide-react';

const serviceOptions: SelectableOption[] = [
  {
    id: 'database',
    name: 'Database Service',
    description: 'Managed PostgreSQL database',
    icon: <Database className="w-5 h-5" />
  },
  {
    id: 'storage',
    name: 'Cloud Storage',
    description: 'File storage and CDN',
    icon: <Cloud className="w-5 h-5" />
  },
  {
    id: 'auth',
    name: 'Authentication',
    description: 'User management system',
    icon: <Shield className="w-5 h-5" />
  }
];

function ServiceSelector() {
  const [selectedServices, setSelectedServices] = useState<string[]>([]);

  return (
    <PushButtonSelector
      options={serviceOptions}
      selectedIds={selectedServices}
      onSelectionChange={setSelectedServices}
      title="Select Services"
      helpText="Choose the services you want to enable"
      selectionSummary={true}
      multiSelect={true}
    />
  );
}
```
