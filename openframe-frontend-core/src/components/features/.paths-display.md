A React component for displaying file system paths with copy-to-clipboard functionality, featuring ODS styling and cross-platform path support. Designed for use cases like antivirus exclusions or system configuration paths.

## Key Components

- **PathsDisplay**: Main component that renders a list of paths with optional copy buttons
- **PathsDisplayProps**: TypeScript interface defining component props
- **OPENFRAME_PATHS**: Pre-defined path constants for Windows, macOS, and Linux platforms
- **getOpenFramePaths**: Utility function to retrieve platform-specific OpenFrame paths
- **OpenFramePathsPlatform**: Type union for supported platforms

## Usage Example

```typescript
import { PathsDisplay, getOpenFramePaths } from './paths-display'

// Basic usage with custom paths
const systemPaths = [
  'C:\\Program Files\\MyApp\\',
  'C:\\Users\\%USERNAME%\\AppData\\Local\\MyApp\\'
]

function PathsList() {
  const handleCopyPath = (path: string) => {
    navigator.clipboard.writeText(path)
    toast({ title: 'Copied!', description: 'Path copied to clipboard' })
  }

  return (
    <PathsDisplay
      paths={systemPaths}
      title="Antivirus Exclusions"
      description="Add these paths to your antivirus whitelist"
      onCopyPath={handleCopyPath}
      showCopyButtons={true}
    />
  )
}

// Using pre-defined OpenFrame paths
function OpenFramePaths() {
  const windowsPaths = getOpenFramePaths('windows')
  
  return (
    <PathsDisplay
      paths={windowsPaths}
      title="OpenFrame Installation Paths"
    />
  )
}
```
