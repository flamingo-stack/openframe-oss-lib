A flexible section navigation component that renders a list of selectable sections with support for badges, descriptions, and customizable layouts.

## Key Components

- **SectionItem**: Interface defining section structure with id, title, subtitle, description, badges, and screenshots
- **SectionSelector**: Main component that renders a collection of section buttons
- **SectionButton**: Internal button component handling individual section rendering
- **SectionSelectorProps**: Configuration interface supporting vertical/wrap layouts, responsive widths, and styling options

## Usage Example

```typescript
import { SectionSelector, SectionItem } from './section-selector'

const sections: SectionItem[] = [
  {
    id: 'overview',
    title: 'Project Overview',
    subtitle: 'Get started with your project',
    description: 'Learn the basics and setup requirements',
    number: '1',
    badge: { text: 'New', variant: 'button', colorScheme: 'green' }
  },
  {
    id: 'setup',
    title: 'Environment Setup',
    subtitle: 'Configure your development environment',
    number: '2'
  }
]

export function NavigationExample() {
  const [activeSection, setActiveSection] = useState('overview')

  return (
    <SectionSelector
      sections={sections}
      activeSection={activeSection}
      onSectionChange={setActiveSection}
      layout="vertical"
      buttonWidth="responsive"
      showDescription={true}
    />
  )
}
```
