A comprehensive SEO editor component that provides editable meta tag fields alongside a real-time social media preview, with intelligent fallback handling for auto-populated content.

## Key Components

- **SEOEditorPreview**: Main component with form fields for SEO title, description, keywords, and OG image
- **SEOEditorPreviewProps**: Interface defining required SEO fields, fallback sources, change handlers, and optional upload functionality
- **Live Preview**: Real-time social media card preview showing how content appears when shared
- **Image Upload**: Optional file upload with drag-and-drop interface for OG images
- **Fallback System**: Automatically uses title/summary/featuredImage when SEO fields are empty

## Usage Example

```typescript
import { SEOEditorPreview } from './seo-editor-preview';

function BlogEditor() {
  const [seoData, setSeoData] = useState({
    seoTitle: '',
    seoDescription: '',
    seoKeywords: '',
    ogImageUrl: ''
  });

  const handleImageUpload = async (file: File) => {
    // Upload logic here
    return 'https://example.com/uploaded-image.jpg';
  };

  return (
    <SEOEditorPreview
      seoTitle={seoData.seoTitle}
      seoDescription={seoData.seoDescription}
      seoKeywords={seoData.seoKeywords}
      ogImageUrl={seoData.ogImageUrl}
      title="Blog Post Title"
      summary="Blog post summary..."
      featuredImage="/featured.jpg"
      onSeoTitleChange={(value) => setSeoData(prev => ({ ...prev, seoTitle: value }))}
      onSeoDescriptionChange={(value) => setSeoData(prev => ({ ...prev, seoDescription: value }))}
      onSeoKeywordsChange={(value) => setSeoData(prev => ({ ...prev, seoKeywords: value }))}
      onOgImageUrlChange={(value) => setSeoData(prev => ({ ...prev, ogImageUrl: value }))}
      onOgImageUpload={handleImageUpload}
      domain="myblog.com"
    />
  );
}
```
