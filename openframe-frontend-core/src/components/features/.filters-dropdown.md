This component provides a customizable dropdown filter interface with support for checkboxes, radio buttons, and select options, featuring responsive design and mobile-optimized positioning.

## Key Components

- **`FiltersDropdown`** - Main dropdown component with trigger and filter panel
- **`FilterCheckbox`** - Custom styled checkbox component with checkmark animation
- **`FilterOption`** - Interface for individual filter options with separator support
- **`FilterSection`** - Interface for grouped filter sections with different input types
- **`FiltersDropdownProps`** - Main component props including current filters and callbacks

## Usage Example

```typescript
import { FiltersDropdown, FilterSection } from './filters-dropdown'

const filterSections: FilterSection[] = [
  {
    id: 'status',
    title: 'Status',
    type: 'checkbox',
    allowSelectAll: true,
    options: [
      { id: 'active', label: 'Active', value: 'active' },
      { id: 'inactive', label: 'Inactive', value: 'inactive' }
    ]
  },
  {
    id: 'priority',
    title: 'Priority',
    type: 'radio',
    options: [
      { id: 'high', label: 'High', value: 'high' },
      { id: 'medium', label: 'Medium', value: 'medium' }
    ]
  }
]

function MyComponent() {
  const [appliedFilters, setAppliedFilters] = useState<Record<string, string[]>>({})

  return (
    <FiltersDropdown
      triggerLabel="Filter Items"
      sections={filterSections}
      onApply={setAppliedFilters}
      currentFilters={appliedFilters}
      placement="bottom-start"
    />
  )
}
```
