A unified, accessible filter chip component providing consistent styling and behavior for tags, categories, and interactive filters across the application.

## Key Components

- **FilterChip**: Main component with multiple visual variants (`selected`, `unselected`, `category`, `subcategory`, `tag`, `info`)
- **FilterChipProps**: Interface defining component properties including variant, size, removability, and event handlers
- Built-in accessibility features with ARIA attributes and keyboard navigation support
- Responsive sizing with mobile-optimized touch targets

## Usage Example

```typescript
import { FilterChip } from './components/filter-chip'

function FilterExample() {
  const [selectedFilters, setSelectedFilters] = useState<string[]>([])
  
  const handleRemove = (filterId: string) => {
    setSelectedFilters(prev => prev.filter(id => id !== filterId))
  }

  return (
    <div className="flex gap-2 flex-wrap">
      {/* Category filter */}
      <FilterChip
        id="tech"
        label="Technology"
        variant="category"
        onClick={() => console.log('Category clicked')}
      />
      
      {/* Removable tag */}
      <FilterChip
        id="javascript"
        label="JavaScript"
        variant="tag"
        removable
        onRemove={() => handleRemove('javascript')}
      />
      
      {/* Info chip (non-interactive) */}
      <FilterChip
        id="count"
        label="25 results"
        variant="info"
        size="sm"
      />
    </div>
  )
}
```
