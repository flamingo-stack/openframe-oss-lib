A reusable sticky navigation component for section-based navigation with scroll spy functionality, commonly used in vendor detail pages, knowledge base articles, and documentation.

## Key Components

- **`StickySectionNav`** - Main navigation component with visual ribbon indicator
- **`useSectionNavigation`** - Hook providing scroll spy and click handling logic
- **`StickyNavSection`** - Interface defining navigation section structure

## Usage Example

```typescript
import { StickySectionNav, useSectionNavigation, StickyNavSection } from './sticky-section-nav'

function DocumentationPage() {
  const overviewRef = useRef<HTMLElement>(null)
  const installRef = useRef<HTMLElement>(null)
  
  const sections = [
    { id: 'overview', ref: overviewRef },
    { id: 'installation', ref: installRef }
  ]
  
  const { activeSection, handleSectionClick } = useSectionNavigation(
    sections,
    { offset: 100 }
  )
  
  const navSections: StickyNavSection[] = [
    { id: 'overview', label: 'Overview' },
    { id: 'installation', label: 'Installation' }
  ]

  return (
    <div className="flex">
      <StickySectionNav
        sections={navSections}
        activeSection={activeSection}
        onSectionClick={handleSectionClick}
        ribbonPosition="left"
        ribbonColor="#FFC008"
      />
      <main>
        <section ref={overviewRef} id="overview">...</section>
        <section ref={installRef} id="installation">...</section>
      </main>
    </div>
  )
}
```
