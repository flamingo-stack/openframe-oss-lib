A reusable sticky navigation component that provides section-based navigation with visual indicators and automatic scroll spy functionality, commonly used for page sections in vendor details, documentation, and knowledge base pages.

## Key Components

- **`StickySectionNav`** - Main navigation component with customizable ribbon positioning and colors
- **`StickyNavSection`** - Interface defining section structure with id and label
- **`useSectionNavigation`** - Hook that handles scroll spy detection and smooth scrolling to sections
- **`StickySectionNavProps`** - Configuration interface for ribbon position, colors, and styling

## Usage Example

```typescript
import { StickySectionNav, useSectionNavigation } from './sticky-section-nav'
import { useRef } from 'react'

function DocumentationPage() {
  const overviewRef = useRef<HTMLElement>(null)
  const featuresRef = useRef<HTMLElement>(null)
  const apiRef = useRef<HTMLElement>(null)

  const sections = [
    { id: 'overview', ref: overviewRef },
    { id: 'features', ref: featuresRef },
    { id: 'api', ref: apiRef }
  ]

  const { activeSection, handleSectionClick } = useSectionNavigation(sections, {
    offset: 100
  })

  const navSections = [
    { id: 'overview', label: 'Overview' },
    { id: 'features', label: 'Features' },
    { id: 'api', label: 'API Reference' }
  ]

  return (
    <div className="flex">
      <StickySectionNav
        sections={navSections}
        activeSection={activeSection}
        onSectionClick={handleSectionClick}
        ribbonPosition="left"
        ribbonColor="#FFC008"
      />
      <main>
        <section ref={overviewRef} id="overview">Overview content</section>
        <section ref={featuresRef} id="features">Features content</section>
        <section ref={apiRef} id="api">API content</section>
      </main>
    </div>
  )
}
```
