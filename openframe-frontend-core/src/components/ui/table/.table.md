A comprehensive React table component that provides a flexible, responsive data table with advanced features like sorting, filtering, selection, pagination, and row actions for complex data display scenarios.

## Key Components

- **Table**: Main table component with generic type support for type-safe data rendering
- **injectActionsColumn**: Utility function that dynamically adds an actions column when row actions are provided
- **Row selection**: Built-in checkbox selection with bulk actions support
- **Responsive design**: Automatic desktop/mobile layout switching with customizable mobile rendering
- **Loading states**: Skeleton loading with placeholder rows to maintain consistent height
- **Pagination**: Support for both cursor-based and page-based pagination patterns
- **Sorting & filtering**: Column-level sorting and filtering capabilities
- **Empty states**: Customizable empty state messaging

## Usage Example

```typescript
import { Table } from './table'

interface User {
  id: string
  name: string
  email: string
  status: 'active' | 'inactive'
}

const UserTable = () => {
  const [users, setUsers] = useState<User[]>([])
  const [selectedRows, setSelectedRows] = useState<User[]>([])

  const columns: TableColumn<User>[] = [
    {
      key: 'name',
      label: 'Name',
      renderCell: (user) => <strong>{user.name}</strong>,
      sortable: true
    },
    {
      key: 'email',
      label: 'Email',
      align: 'left'
    },
    {
      key: 'status',
      label: 'Status',
      renderCell: (user) => (
        <span className={user.status === 'active' ? 'text-green-500' : 'text-red-500'}>
          {user.status}
        </span>
      )
    }
  ]

  const rowActions: RowAction<User>[] = [
    {
      label: 'Edit',
      onClick: (user) => console.log('Edit user:', user.id),
      variant: 'outline'
    },
    {
      label: 'Delete',
      onClick: (user) => console.log('Delete user:', user.id),
      variant: 'destructive'
    }
  ]

  return (
    <Table
      data={users}
      columns={columns}
      rowKey="id"
      rowActions={rowActions}
      selectable
      selectedRows={selectedRows}
      onSelectionChange={setSelectedRows}
      onRowClick={(user) => console.log('Row clicked:', user)}
      bulkActions={[
        {
          label: 'Delete Selected',
          onClick: (users) => console.log('Bulk delete:', users),
          requiresSelection: true
        }
      ]}
    />
  )
}
```
