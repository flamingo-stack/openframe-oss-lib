A reusable full-screen image gallery modal component with keyboard navigation, thumbnail controls, and error handling for unsupported image formats like HEIC.

## Key Components

- **ImageGalleryModal**: Main component that renders a full-screen modal with image navigation
- **Navigation Controls**: Previous/next buttons with chevron icons (only shown when multiple images exist)
- **Keyboard Support**: Arrow keys for navigation, Escape to close
- **Image Counter**: Shows current position in gallery (e.g., "3 / 10")
- **Error Handling**: Graceful fallback for unsupported formats with download option

## Usage Example

```typescript
import { ImageGalleryModal } from './image-gallery-modal';

function EventPhotos() {
  const [isGalleryOpen, setIsGalleryOpen] = useState(false);
  const [selectedImageIndex, setSelectedImageIndex] = useState(0);
  
  const images = [
    '/photos/event1.jpg',
    '/photos/event2.png',
    '/photos/event3.heic'
  ];

  const openGallery = (index: number) => {
    setSelectedImageIndex(index);
    setIsGalleryOpen(true);
  };

  return (
    <div>
      {images.map((image, index) => (
        <img 
          key={index}
          src={image} 
          onClick={() => openGallery(index)}
          className="cursor-pointer"
        />
      ))}
      
      <ImageGalleryModal
        images={images}
        isOpen={isGalleryOpen}
        onClose={() => setIsGalleryOpen(false)}
        initialIndex={selectedImageIndex}
      />
    </div>
  );
}
```
