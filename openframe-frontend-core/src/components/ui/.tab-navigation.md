A flexible tab navigation component for Next.js applications that supports both controlled state management and automatic URL synchronization for preserving tab state across page reloads and navigation.

## Key Components

- **`TabNavigation`** - Main component with dual operation modes (controlled vs URL-synced)
- **`TabItem`** - Interface defining tab structure with icon, label, component, and alert properties
- **`TabNavigationUrlSyncOptions`** - Configuration for URL synchronization behavior
- **`getTabById`** - Utility function to find a tab by its ID
- **`getTabComponent`** - Utility function to extract a tab's associated component

## Usage Example

```typescript
import { TabNavigation, TabItem } from './tab-navigation'
import { HomeIcon, SettingsIcon, ProfileIcon } from './icons'

const tabs: TabItem[] = [
  {
    id: 'dashboard',
    label: 'Dashboard',
    icon: HomeIcon,
    component: DashboardComponent
  },
  {
    id: 'settings',
    label: 'Settings',
    icon: SettingsIcon,
    component: SettingsComponent,
    hasAlert: true,
    alertType: 'warning'
  }
]

// URL-synced mode (recommended)
function App() {
  return (
    <TabNavigation
      tabs={tabs}
      urlSync={{ paramName: 'view', replaceState: false }}
      defaultTab="dashboard"
    >
      {(activeTab) => {
        const TabComponent = getTabComponent(tabs, activeTab)
        return TabComponent ? <TabComponent /> : null
      }}
    </TabNavigation>
  )
}

// Controlled mode
function ControlledApp() {
  const [activeTab, setActiveTab] = useState('dashboard')
  
  return (
    <TabNavigation
      tabs={tabs}
      activeTab={activeTab}
      onTabChange={setActiveTab}
      urlSync={false}
    />
  )
}
```
