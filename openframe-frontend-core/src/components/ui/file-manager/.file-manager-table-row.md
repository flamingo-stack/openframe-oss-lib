A React component that renders a single row in a file manager table interface, displaying file/folder information with interactive capabilities like selection, context menus, and navigation.

## Key Components

- **FileManagerTableRow**: Main component that renders file/folder data in a table row format
- **Interactive Elements**: Checkbox for selection, context menu button, and folder navigation chevron
- **File Information Display**: Shows file icon, name, optional path, size, and modification date
- **Event Handlers**: Manages clicks, double-clicks, context menu actions, and selection state

## Usage Example

```typescript
import { FileManagerTableRow } from './file-manager-table-row'

function FileList() {
  const [selectedFiles, setSelectedFiles] = useState<string[]>([])

  const handleFileSelect = (fileId: string, checked: boolean) => {
    if (checked) {
      setSelectedFiles(prev => [...prev, fileId])
    } else {
      setSelectedFiles(prev => prev.filter(id => id !== fileId))
    }
  }

  return (
    <div>
      <FileManagerTableRow
        file={{
          id: '1',
          name: 'document.pdf',
          type: 'file',
          size: '2.4 MB',
          modified: '2024-01-15'
        }}
        isSelected={selectedFiles.includes('1')}
        showCheckbox={true}
        showPath={false}
        onSelect={(checked) => handleFileSelect('1', checked)}
        onClick={() => console.log('File clicked')}
        onDoubleClick={() => console.log('File opened')}
        onContextMenu={(e) => console.log('Context menu')}
        onActionClick={(action) => console.log('Action:', action)}
      />
    </div>
  )
}
```
