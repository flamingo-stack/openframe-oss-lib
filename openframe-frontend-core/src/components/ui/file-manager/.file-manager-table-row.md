A React component that renders a single row in a file manager table, displaying file/folder information with interactive elements like checkboxes, context menus, and navigation controls.

## Key Components

- **FileManagerTableRow**: Main component that renders a table row with file metadata
- **Interactive Elements**: Checkbox for selection, context menu button, and folder navigation chevron
- **File Display**: Shows file icon, name, optional path, size, and modification date
- **Event Handlers**: Supports click, double-click, context menu, and selection events

## Usage Example

```typescript
import { FileManagerTableRow } from './file-manager-table-row'

function FileList() {
  const [selectedFiles, setSelectedFiles] = useState<string[]>([])
  
  const handleFileSelect = (fileId: string, selected: boolean) => {
    setSelectedFiles(prev => 
      selected 
        ? [...prev, fileId] 
        : prev.filter(id => id !== fileId)
    )
  }

  return (
    <div>
      <FileManagerTableRow
        file={{
          name: "document.pdf",
          type: "file",
          size: "2.5 MB",
          modified: "2024-01-15",
          path: "/documents"
        }}
        isSelected={selectedFiles.includes("document.pdf")}
        showCheckbox={true}
        showPath={true}
        onSelect={(checked) => handleFileSelect("document.pdf", checked)}
        onClick={() => console.log('File clicked')}
        onDoubleClick={() => console.log('File opened')}
        onActionClick={(action) => console.log('Action:', action)}
      />
    </div>
  )
}
```
