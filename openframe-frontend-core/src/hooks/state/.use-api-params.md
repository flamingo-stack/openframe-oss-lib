A React hook that provides type-safe URL state management for REST API parameters without GraphQL dependencies. Handles URL synchronization, type coercion, and provides a clean API for managing search parameters.

## Key Components

- `useApiParams<TSchema>()` - Main hook for URL state management
- `ParamSchema` - Type for defining parameter configurations
- `defineParamSchema()` - Helper for creating typed schemas
- `ParamConfig<T>` - Configuration interface for individual parameters
- `UseApiParamsReturn<TSchema>` - Strongly typed return interface

## Usage Example

```typescript
import { useApiParams, defineParamSchema } from './use-api-params'

// Define your parameter schema
const searchSchema = defineParamSchema({
  search: { type: 'string', default: '' },
  page: { type: 'number', default: 1 },
  tags: { type: 'array', default: [] },
  sortBy: { type: 'string', default: 'name' }
})

function SearchComponent() {
  const { params, setParam, setParams, urlSearchParams } = useApiParams(searchSchema)

  // params is fully typed: { search: string, page: number, tags: string[], sortBy: string }
  
  // Fetch data using URLSearchParams
  const fetchData = async () => {
    const response = await fetch(`/api/items?${urlSearchParams}`)
    return response.json()
  }

  // Type-safe parameter updates
  const handleSearch = (query: string) => {
    setParam('search', query)
    setParam('page', 1) // Reset to first page
  }

  const handleMultipleUpdates = () => {
    setParams({
      search: 'laptop',
      tags: ['electronics', 'sale'],
      page: 1
    })
  }

  return (
    <div>
      <input 
        value={params.search} 
        onChange={(e) => handleSearch(e.target.value)}
        placeholder="Search..."
      />
      <div>Page: {params.page}</div>
      <div>Tags: {params.tags.join(', ')}</div>
    </div>
  )
}
```

The hook automatically handles URL synchronization, type coercion from strings to the correct types, and preserves other URL parameters not managed by the schema.
