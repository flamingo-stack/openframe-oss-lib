A React hook for managing URL search parameters with type safety and automatic synchronization for REST API integrations. Provides manual schema definition without GraphQL dependencies.

## Key Components

- **`useApiParams<TParams>`** - Main hook that manages URL state with type coercion and automatic synchronization
- **`ParamSchema`** - Interface for defining parameter schemas with type, default value, and required flags
- **`UseApiParamsReturn<TParams>`** - Return type providing params object, URLSearchParams, and update methods
- **`createSearchParams`** - Helper function to convert parameter objects to URLSearchParams
- **`UseApiParamsOptions`** - Configuration interface with debug logging option

## Usage Example

```typescript
import { useApiParams } from './use-api-params'

// Define parameter schema
const schema = {
  search: { type: 'string' as const, default: '' },
  page: { type: 'number' as const, default: 1 },
  tags: { type: 'array' as const, default: [] },
  active: { type: 'boolean' as const, default: true }
}

function ItemsPage() {
  const { params, urlSearchParams, setParam, setParams } = useApiParams(schema)
  
  // Use with fetch API
  const fetchItems = async () => {
    const response = await fetch(`/api/items?${urlSearchParams}`)
    return response.json()
  }
  
  // Update individual parameters
  const handleSearch = (query: string) => {
    setParam('search', query)
    setParam('page', 1) // Reset page on new search
  }
  
  // Update multiple parameters
  const handleFilter = () => {
    setParams({
      tags: ['electronics', 'sale'],
      active: true,
      page: 1
    })
  }
  
  return (
    <div>
      <input 
        value={params.search} 
        onChange={(e) => handleSearch(e.target.value)} 
      />
      <p>Page: {params.page}</p>
      <p>Tags: {params.tags.join(', ')}</p>
    </div>
  )
}
```
