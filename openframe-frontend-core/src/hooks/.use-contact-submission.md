A React hook that manages contact form submission with loading states, toast notifications, and optional post-submission actions like redirects or callbacks.

## Key Components

- **`useContactSubmission`** - Main hook that handles form submission logic
- **`ContactSubmissionOptions`** - Configuration interface for customizing behavior
- **`ContactFormData`** - Type definition for contact form fields
- **`submit`** - Function to submit form data to `/api/contact` endpoint
- **State management** - `isSubmitting`, `isSuccess` states and automatic redirect handling

## Usage Example

```typescript
import { useContactSubmission } from './use-contact-submission';

function ContactForm() {
  const { submit, isSubmitting, isSuccess } = useContactSubmission({
    userId: 'user123',
    successRedirectUrl: '/thank-you',
    successToastMessage: 'We\'ll get back to you soon!',
    onSuccess: () => console.log('Form submitted successfully')
  });

  const handleSubmit = async (formData: ContactFormData) => {
    try {
      await submit({
        name: 'John Doe',
        email: 'john@example.com',
        companySize: '10-50',
        referralSource: 'Google',
        helpCategory: 'Support',
        message: 'Need help with integration'
      });
    } catch (error) {
      // Handle submission error
    }
  };

  return (
    <form onSubmit={(e) => { /* handle form submission */ }}>
      <button type="submit" disabled={isSubmitting}>
        {isSubmitting ? 'Sending...' : 'Send Message'}
      </button>
    </form>
  );
}
```
