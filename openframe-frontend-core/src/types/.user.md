This file defines TypeScript types for user-related data structures in the application. It provides type definitions for user profiles, favorites, activities, and preferences to ensure type safety across the codebase.

## Key Components

- **`UserProfile`** - Complete user profile with personal and professional information
- **`UserFavorite`** - User's favorited vendors with timestamps
- **`UserActivity`** - Tracking of user interactions and system events
- **`UserPreferences`** - User-specific application settings and preferences

## Usage Example

```typescript
import { UserProfile, UserActivity, UserPreferences } from './user'

// Creating a user profile
const user: UserProfile = {
  id: "uuid-123",
  full_name: "John Doe",
  email: "john@example.com",
  avatar_url: "https://example.com/avatar.jpg",
  job_title: "Software Engineer",
  company: "Tech Corp",
  bio: "Experienced developer",
  employee_id: "EMP001",
  department: "Engineering",
  role: "user",
  msp_id: "msp-456",
  created_at: "2024-01-01T00:00:00Z",
  updated_at: "2024-01-01T00:00:00Z"
}

// Tracking user activity
const activity: UserActivity = {
  id: 1,
  user_id: user.id,
  vendor_id: 123,
  activity_type: "viewed",
  created_at: "2024-01-01T00:00:00Z",
  metadata: { page: "/vendor/123" }
}

// User preferences
const preferences: UserPreferences = {
  id: "pref-789",
  user_id: user.id,
  theme: "dark",
  email_notifications: true,
  created_at: "2024-01-01T00:00:00Z",
  updated_at: "2024-01-01T00:00:00Z"
}
```
