Defines TypeScript interfaces for category-related components and data structures used throughout the application's category management system.

## Key Components

**CategoryCardProps** - Main interface for category card components with comprehensive category metadata including counts and display options

**RealCategoryCardProps** - Enhanced category card interface with loading states and vendor data for real-time category displays

**CategoryData** - Data container interface for category collections with loading states and error handling

## Usage Example

```typescript
import { CategoryCardProps, RealCategoryCardProps, CategoryData } from './category';

// Basic category card usage
const categoryCard: CategoryCardProps = {
  category: {
    id: "tech-001",
    name: "Technology",
    slug: "technology",
    description: "Latest tech products and services",
    icon_url: "/icons/tech.svg",
    iconUrls: ["/icons/tech.svg", "/icons/tech-alt.svg"],
    subcategory_count: 15,
    productCount: 150
  },
  vendorCount: 25,
  className: "featured-category"
};

// Real-time category with loading states
const realCategoryCard: RealCategoryCardProps = {
  category: {
    id: "tech-001",
    name: "Technology", 
    slug: "technology",
    icon_url: "/icons/tech.svg"
  },
  isLoading: false,
  isLoadingVendorCount: true,
  vendors: []
};

// Category data collection
const categoryData: CategoryData = {
  data: [/* category objects */],
  isLoading: false,
  error: null,
  vendors: [],
  loading: false
};
```
